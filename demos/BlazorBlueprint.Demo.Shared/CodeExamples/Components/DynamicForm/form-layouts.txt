private readonly FormSchema schema = new()
{
    Fields =
    [
        new() { Name = "Name", Label = "Name", Type = FieldType.Text },
        new() { Name = "Email", Label = "Email", Type = FieldType.Email },
        new() { Name = "Role", Label = "Role", Type = FieldType.Select,
            Options = [new("dev", "Developer"), new("design", "Designer")] }
    ]
};

// Vertical (default) — labels above fields
<BbDynamicForm Schema="@schema" @bind-Values="values" Layout="FormLayout.Vertical" />

// Horizontal — labels to the left of fields (LabelWidth aligns inputs)
<BbDynamicForm Schema="@schema" @bind-Values="values" Layout="FormLayout.Horizontal" LabelWidth="4rem" />

// Inline — fields arranged in a single row
<BbDynamicForm Schema="@schema" @bind-Values="values" Layout="FormLayout.Inline" />
