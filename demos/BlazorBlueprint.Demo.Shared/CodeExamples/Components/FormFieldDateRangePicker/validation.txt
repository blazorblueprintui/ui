<BbFormFieldDateRangePicker @bind-Value="dateRange"
                           Label="Booking dates"
                           HelperText="Required. Select your check-in and check-out dates."
                           ErrorText="@errorMessage" />
<BbButton OnClick="Validate">Submit</BbButton>

@code {
    private DateRange? dateRange;
    private string? errorMessage;

    private void Validate()
    {
        errorMessage = dateRange is null
            ? "Please select a date range."
            : null;
    }
}
