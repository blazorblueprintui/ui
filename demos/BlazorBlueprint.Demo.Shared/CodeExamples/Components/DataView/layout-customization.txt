@* ── GridClass: override columns and gap ──────────────────────────────────────
   GridClass is merged on top of the default via TailwindMerge, so only the
   classes you supply that conflict with a default override it — everything
   else is preserved.
   In templates, use @product.Description (from MockDataService) for card body text. *@

@* 2-column grid with wider gap *@
<BbDataView TItem="Product" Data="@products"
            GridClass="grid-cols-1 sm:grid-cols-2 gap-6">
    <GridTemplate Context="product">
        ...
    </GridTemplate>
</BbDataView>

@* 4-column responsive grid with tighter gap *@
<BbDataView TItem="Product" Data="@products"
            GridClass="grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-3">
    <GridTemplate Context="product">
        ...
    </GridTemplate>
</BbDataView>

@* ── ListClass: override gap and/or add dividers ──────────────────────────────
   ListClass is merged the same way on top of the default flex flex-col gap-2. *@

@* Wider spacing between rows *@
<BbDataView TItem="Person" Data="@people" ListClass="gap-4">
    <ListTemplate Context="person">
        ...
    </ListTemplate>
</BbDataView>

@* Compact divider list — no gap, border between each row *@
<BbDataView TItem="Person" Data="@people" ListClass="gap-0 divide-y">
    <ListTemplate Context="person">
        ...
    </ListTemplate>
</BbDataView>
